// JavaScript - Install dengan npm atau CDN

        const response = await axios.get(
            'https://ziferr-tiktok-api.vercel.app/api/tiktok',
            { params: { url } }

        console.error('❌ Error:', error.message);
    }
}

Future> downloadTikTok(String url) async {

            Uri.parse(
                'https://ziferr-tiktok-api.vercel.app/api/tiktok?url=${Uri.encodeComponent(url)}'

            final data = json.decode(response.body);
            print('✅ Video: ${data['video']}');
            print('✅ Audio: ${data['audio']}');
            print('✅ Caption: ${data['caption']}');
            print('✅ Author: ${data['author']}');
            return data;
        }
    } catch (e) {
        print('❌ Error: $e');
    }
}

<?php
// PHP - Menggunakan cURL

function downloadTikTok($url) {
    $apiUrl = "https://ziferr-tiktok-api.vercel.app/api/tiktok?url=" . urlencode($url);
    
    $ch = curl_init();
    curl_setopt($ch, CURLOPT_URL, $apiUrl);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
    curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
    curl_setopt($ch, CURLOPT_TIMEOUT, 30);
    
    $response = curl_exec($ch);
    $error = curl_error($ch);
    curl_close($ch);
    
    if ($error) {
        die("❌ Error: " . $error);
    }
    
    $data = json_decode($response, true);
    
    if ($data['success']) {
        echo "✅ Video: " . $data['video'] . "\n";
        echo "✅ Audio: " . $data['audio'] . "\n";
        echo "✅ Caption: " . $data['caption'] . "\n";
        echo "✅ Author: " . $data['author'] . "\n";
        return $data;
    }
}

// Penggunaan
downloadTikTok('https://vt.tiktok.com/ZSCUBjoFb');
?>

// Java - Menggunakan HttpClient (Java 11+)

import java.net.URI;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.net.URLEncoder;
import java.nio.charset.StandardCharsets;

public class TikTokDownloader {
    public static void main(String[] args) throws Exception {
        String url = "https://vt.tiktok.com/ZSCUBjoFb";
        String encodedUrl = URLEncoder.encode(url, StandardCharsets.UTF_8);
        
        HttpClient client = HttpClient.newHttpClient();
        HttpRequest request = HttpRequest.newBuilder()
            .uri(URI.create("https://ziferr-tiktok-api.vercel.app/api/tiktok?url=" + encodedUrl))
            .GET()
            .build();
            
        HttpResponse response = client.send(request, 
            HttpResponse.BodyHandlers.ofString());
            
        System.out.println("Response: " + response.body());
    }
}

package main

import (
    "fmt"
    "net/url"
    "github.com/go-resty/resty/v2"
)

func main() {
    client := resty.New()
    
    url := "https://vt.tiktok.com/ZSCUBjoFb"
    encodedUrl := url.QueryEscape(url)
    
    resp, err := client.R().
        Get("https://ziferr-tiktok-api.vercel.app/api/tiktok?url=" + encodedUrl)
    
    if err != nil {
        fmt.Println("❌ Error:", err)
        return
    }
    
    fmt.Println("✅ Response:", resp.String())
}
